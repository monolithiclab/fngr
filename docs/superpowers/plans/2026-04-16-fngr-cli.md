# fngr CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Go CLI tool for logging and tracking events with hierarchical parent-child relationships, unified key-value metadata, and FTS5-powered search.

**Architecture:** Kong-based CLI with SQLite storage. Events have free-form text and unified metadata (author, tags, people, arbitrary key=value) stored in a separate `event_meta` table. An FTS5 virtual table indexes both body text and metadata for fast filtered queries. Parent-child trees use recursive CTEs.

**Tech Stack:** Go 1.26, Kong (CLI), modernc.org/sqlite (pure Go, no CGO), SQLite FTS5

**Spec:** `docs/superpowers/specs/2026-04-16-fngr-cli-design.md`

```go
import (
	"fmt"

	"github.com/alecthomas/kong"

var version = "dev"

type CLI struct {
	DB string `help:"Path to database file." env:"FNGR_DB" type:"path"`

	Add    internal.AddCmd    `cmd:"" default:"withargs" help:"Add an event (default command)."`
	List   internal.ListCmd   `cmd:"" help:"List events."`
	Show   internal.ShowCmd   `cmd:"" help:"Show a single event."`
	Delete internal.DeleteCmd `cmd:"" help:"Delete an event."`
	Meta   internal.MetaCmd   `cmd:"" help:"List all metadata keys and values."`
}

func main() {
	var cli CLI
	ctx := kong.Parse(&cli,
		kong.Name("fngr"),
		kong.Description("A CLI to log and track events."),
		kong.Vars{"version": version},
		kong.UsageOnError(),
	)

	dbPath, err := internal.ResolveDBPath(cli.DB)
	if err != nil {
		fmt.Fprintf(os.Stderr, "error: %v\n", err)
		os.Exit(1)
	}

	db, err := internal.OpenDB(dbPath, ctx.Command() == "add")
	if err != nil {
		fmt.Fprintf(os.Stderr, "error: %v\n", err)
		os.Exit(1)
	}
	defer db.Close()

	err = ctx.Run(db)
	ctx.FatalIfErrorf(err)
}
```

```go
package internal

import (
	"database/sql"
	"fmt"
	"os"
	"path/filepath"

	_ "modernc.org/sqlite"
)

func ResolveDBPath(explicit string) (string, error) {
	if explicit != "" {
		return explicit, nil
	}
	if _, err := os.Stat(".fngr.db"); err == nil {
		return ".fngr.db", nil
	}
	home, err := os.UserHomeDir()
	if err != nil {
		return "", fmt.Errorf("cannot determine home directory: %w", err)
	}
	return filepath.Join(home, ".fngr.db"), nil
}

func OpenDB(path string, create bool) (*sql.DB, error) {
	if !create {
		if _, err := os.Stat(path); os.IsNotExist(err) {
			return nil, fmt.Errorf("database not found: %s (use 'fngr add' to create one)", path)
		}
	}

	db, err := sql.Open("sqlite", path)
	if err != nil {
		return nil, fmt.Errorf("cannot open database: %w", err)
	}

	if _, err := db.Exec("PRAGMA foreign_keys = ON"); err != nil {
		db.Close()
		return nil, fmt.Errorf("cannot enable foreign keys: %w", err)
	}

	if err := initSchema(db); err != nil {
		db.Close()
		return nil, fmt.Errorf("cannot initialize schema: %w", err)
	}

	return db, nil
}
```

```go
func initSchema(db *sql.DB) error {
	schema := `
		CREATE TABLE IF NOT EXISTS events (
			id         INTEGER PRIMARY KEY AUTOINCREMENT,
			parent_id  INTEGER REFERENCES events(id) ON DELETE CASCADE,
			text       TEXT NOT NULL,
			created_at DATETIME NOT NULL DEFAULT (datetime('now'))
		);

		CREATE TABLE IF NOT EXISTS event_meta (
			event_id INTEGER NOT NULL REFERENCES events(id) ON DELETE CASCADE,
			key      TEXT NOT NULL,
			value    TEXT NOT NULL
		);

		CREATE INDEX IF NOT EXISTS idx_event_meta_key_value ON event_meta(key, value);
		CREATE INDEX IF NOT EXISTS idx_event_meta_event_id ON event_meta(event_id);

		CREATE VIRTUAL TABLE IF NOT EXISTS events_fts USING fts5(
			content,
			tokenize = 'unicode61 tokenchars "=/"'
		);

		CREATE TRIGGER IF NOT EXISTS trg_events_fts_delete
		AFTER DELETE ON events
		BEGIN
			DELETE FROM events_fts WHERE rowid = OLD.id;
		END;
	`
	_, err := db.Exec(schema)
	return err
}
```

Note: The FTS insert trigger is not created here because it needs to fire after both the event and its metadata are inserted. FTS content is built in the `AddEvent` function (Task 4) within a transaction.

```go
package internal

import "regexp"

var (
	hashTagRe = regexp.MustCompile(`#([\w][\w/\-]*)`)
	atTagRe   = regexp.MustCompile(`@([\w][\w/\-]*)`)
)

type Meta struct {
	Key   string
	Value string
}

func ParseBodyTags(text string) []Meta {
	seen := make(map[Meta]bool)
	var result []Meta

	for _, match := range atTagRe.FindAllStringSubmatch(text, -1) {
		m := Meta{Key: "people", Value: match[1]}
		if !seen[m] {
			seen[m] = true
			result = append(result, m)
		}
	}

	for _, match := range hashTagRe.FindAllStringSubmatch(text, -1) {
		m := Meta{Key: "tag", Value: match[1]}
		if !seen[m] {
			seen[m] = true
			result = append(result, m)
		}
	}

	return result
}
```

```go
func ParseFlagMeta(flags []string) ([]Meta, error) {
	var result []Meta
	for _, f := range flags {
		key, value, ok := strings.Cut(f, "=")
		if !ok {
			return nil, fmt.Errorf("invalid metadata format %q: expected key=value", f)
		}
		result = append(result, Meta{Key: key, Value: value})
	}
	return result, nil
}
```

```go
func CollectMeta(text string, flags []string, author string) ([]Meta, error) {
	seen := make(map[Meta]bool)
	var result []Meta

	add := func(m Meta) {
		if !seen[m] {
			seen[m] = true
			result = append(result, m)
		}
	}

	add(Meta{Key: "author", Value: author})

	for _, m := range ParseBodyTags(text) {
		add(m)
	}

	flagMeta, err := ParseFlagMeta(flags)
	if err != nil {
		return nil, err
	}
	for _, m := range flagMeta {
		add(m)
	}

	return result, nil
}
```

```go
func BuildFTSContent(text string, meta []Meta) string {
	var b strings.Builder
	b.WriteString(text)
	for _, m := range meta {
		b.WriteByte(' ')
		b.WriteString(m.Key)
		b.WriteByte('=')
		b.WriteString(m.Value)
	}
	return b.String()
}
```

```go
package internal

import "strings"

func PreprocessFilter(expr string) string {
	if expr == "" {
		return ""
	}

	expr = strings.TrimSpace(expr)
	tokens := tokenizeFilter(expr)
	var parts []string

	for i := 0; i < len(tokens); i++ {
		tok := tokens[i]
		switch tok {
		case "&":
			parts = append(parts, "AND")
		case "|":
			parts = append(parts, "OR")
		default:
			negated := false
			if strings.HasPrefix(tok, "!") {
				negated = true
				tok = tok[1:]
			}

			var term string
			if strings.HasPrefix(tok, "#") {
				term = `"tag=` + tok[1:] + `"`
			} else if strings.HasPrefix(tok, "@") {
				term = `"people=` + tok[1:] + `"`
			} else if strings.Contains(tok, "=") {
				term = `"` + tok + `"`
			} else {
				term = tok
			}

			if negated {
				parts = append(parts, "NOT", term)
			} else {
				parts = append(parts, term)
			}
		}
	}

	return strings.Join(parts, " ")
}

func tokenizeFilter(expr string) []string {
	var tokens []string
	var current strings.Builder

	flush := func() {
		s := strings.TrimSpace(current.String())
		if s != "" {
			tokens = append(tokens, s)
		}
		current.Reset()
	}

	for _, ch := range expr {
		switch ch {
		case '&', '|':
			flush()
			tokens = append(tokens, string(ch))
		case ' ', '\t':
			flush()
		default:
			current.WriteRune(ch)
		}
	}
	flush()

	return tokens
}
```

```go
type ListOpts struct {
	Filter string
	From   string
	To     string
}

func ListEvents(db *sql.DB, opts ListOpts) ([]Event, error) {
	var query string
	var args []any

	if opts.Filter != "" {
		ftsQuery := PreprocessFilter(opts.Filter)
		query = `
			SELECT e.id, e.parent_id, e.text, e.created_at
			FROM events e
			JOIN events_fts f ON f.rowid = e.id
			WHERE events_fts MATCH ?`
		args = append(args, ftsQuery)
	} else {
		query = `SELECT id, parent_id, text, created_at FROM events WHERE 1=1`
	}

	if opts.From != "" {
		query += " AND created_at >= ?"
		args = append(args, opts.From)
	}
	if opts.To != "" {
		query += " AND created_at <= datetime(?, '+1 day')"
		args = append(args, opts.To)
	}

	query += " ORDER BY created_at ASC"

	rows, err := db.Query(query, args...)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var events []Event
	for rows.Next() {
		var ev Event
		if err := rows.Scan(&ev.ID, &ev.ParentID, &ev.Text, &ev.CreatedAt); err != nil {
			return nil, err
		}
		events = append(events, ev)
	}
	if err := rows.Err(); err != nil {
		return nil, err
	}

	// Load metadata for each event
	for i := range events {
		meta, err := loadMeta(db, events[i].ID)
		if err != nil {
			return nil, err
		}
		events[i].Meta = meta
	}

	return events, nil
}

func loadMeta(db *sql.DB, eventID int64) ([]Meta, error) {
	rows, err := db.Query(
		"SELECT key, value FROM event_meta WHERE event_id = ? ORDER BY key, value",

		return nil, err
	}
	defer rows.Close()

	var meta []Meta
	for rows.Next() {
		var m Meta
		if err := rows.Scan(&m.Key, &m.Value); err != nil {
			return nil, err
		}
		meta = append(meta, m)
	}
	return meta, rows.Err()
```

```go
package internal

import (
	"fmt"
	"strings"
)

func RenderTree(events []Event) string {
	if len(events) == 0 {
		return ""
	}

	// Build parent-to-children index
	children := make(map[int64][]int) // parentID -> indices into events
	idToIndex := make(map[int64]int)
	var roots []int

	for i, ev := range events {
		idToIndex[ev.ID] = i
		if ev.ParentID == nil {
			roots = append(roots, i)
		} else {
			children[*ev.ParentID] = append(children[*ev.ParentID], i)
		}
	}

	var b strings.Builder
	for _, ri := range roots {
		renderNode(&b, events, children, ri, "")
	}

	return b.String()
}

func renderNode(b *strings.Builder, events []Event, children map[int64][]int, idx int, prefix string) {
	ev := events[idx]
	author := metaValue(ev.Meta, "author")

	fmt.Fprintf(b, "%s%-4d%s  %s  %s\n",
		prefix,
		ev.ID,
		ev.CreatedAt.Format("2006-01-02"),
		author,
		ev.Text,
	)

	kids := children[ev.ID]
	for i, childIdx := range kids {
		isLast := i == len(kids)-1

		var childPrefix, nextPrefix string
		if isLast {
			childPrefix = prefix + "└─ "
			nextPrefix = prefix + "   "
		} else {
			childPrefix = prefix + "├─ "
			nextPrefix = prefix + "│  "
		}

		renderNodeChild(b, events, children, childIdx, childPrefix, nextPrefix)
	}
}

func renderNodeChild(b *strings.Builder, events []Event, children map[int64][]int, idx int, prefix string, nextPrefix string) {
	ev := events[idx]
	author := metaValue(ev.Meta, "author")

	fmt.Fprintf(b, "%s%-4d%s  %s  %s\n",
		prefix,
		ev.ID,
		ev.CreatedAt.Format("2006-01-02"),
		author,
		ev.Text,
	)

	kids := children[ev.ID]
	for i, childIdx := range kids {
		isLast := i == len(kids)-1

		var childPrefix, childNextPrefix string
		if isLast {
			childPrefix = nextPrefix + "└─ "
			childNextPrefix = nextPrefix + "   "
		} else {
			childPrefix = nextPrefix + "├─ "
			childNextPrefix = nextPrefix + "│  "
		}

		renderNodeChild(b, events, children, childIdx, childPrefix, childNextPrefix)
	}
}

func metaValue(meta []Meta, key string) string {
	for _, m := range meta {
		if m.Key == key {
			return m.Value
		}
	}
	return ""
}
```

```go
func RenderCSV(events []Event) string {
	var b strings.Builder
	w := csv.NewWriter(&b)

	w.Write([]string{"id", "parent_id", "created_at", "author", "text"})

	for _, ev := range events {
		parentID := ""
		if ev.ParentID != nil {
			parentID = fmt.Sprintf("%d", *ev.ParentID)

			author,
			ev.Text,
		})
	}

	w.Flush()
	return b.String()
```

```go
func GetSubtree(db *sql.DB, rootID int64) ([]Event, error) {
	// Verify root exists
	if _, err := GetEvent(db, rootID); err != nil {
		return nil, err
	}

	rows, err := db.Query(`
		WITH RECURSIVE subtree AS (
			SELECT id, parent_id, text, created_at FROM events WHERE id = ?
			UNION ALL
			SELECT e.id, e.parent_id, e.text, e.created_at
			FROM events e
			JOIN subtree s ON e.parent_id = s.id
		)
		SELECT id, parent_id, text, created_at FROM subtree
		ORDER BY created_at ASC
	`, rootID)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var events []Event
	for rows.Next() {
		var ev Event
		if err := rows.Scan(&ev.ID, &ev.ParentID, &ev.Text, &ev.CreatedAt); err != nil {
			return nil, err
		}
		events = append(events, ev)
	}
	if err := rows.Err(); err != nil {
		return nil, err
	}

	for i := range events {
		meta, err := loadMeta(db, events[i].ID)
		if err != nil {
			return nil, err
		}
		events[i].Meta = meta
	}

	return events, nil
}
```
